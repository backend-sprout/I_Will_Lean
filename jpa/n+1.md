                .mapToObj(i -> "teamName" + i)

                .collect(Collectors.toList());

        users = new ArrayList<>();
        for (int i = 0; i < 10; i++) {
            for (int j = 0; j < 10; j++) {
                final User user = new User("username" + i + "" + j, teams.get(i));

            }
        }
        userRepository.saveAll(users);

        teamRepository.flush();
        userRepository.flush();

    }
